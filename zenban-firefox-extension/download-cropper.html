Download CropperJS

Open this page and save the file as cropper.min.js in the extension folder:

Download CropperJS 1.6.1

Or run this in the console:

fetch('https://cdnjs.cloudflare.com/ajax/libs/cropperjs/1.6.1/cropper.min.js')
    .then(r => r.text())
    .then(text => {
        const a = document.createElement('a');
        a.href = 'data:text/javascript;charset=utf-8,' + encodeURIComponent(text);
        a.download = 'cropper.min.js';
        a.click();
    });
